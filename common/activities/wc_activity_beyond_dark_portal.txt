activity_beyond_dark_portal = {
	expiration_days = 730
	has_activity_window = yes
	days_until_auto_start = 60
	frame = 1
	color = { 0.253906 0.867188 0.105469 1 }
	valid = {
		always = yes
	}
	on_spawn = {
		activity_owner = {
			add_to_temporary_list = summoned_rulers
			every_ally = {
				add_to_temporary_list = summoned_rulers
			}
		}
		# Summons players under summoned rulers
		every_player = {
			limit = {
				NOT = { is_in_list = summoned_rulers }
				any_liege_or_above = { is_in_list = summoned_rulers }
			}
			add_to_temporary_list = summoned_rulers
		}
		
		every_in_list = {
			list = summoned_rulers
			limit = {
				draenor_culture_trigger = no
				draenor_faith_trigger = no
			}
			
			scope:activity = {
				invite_character_to_activity = prev
			}
			
			trigger_event = WCDRA.1000
		}
	}
	on_activate = {
		beyond_dark_portal_find_leader_effect = yes
		every_participant = {
			limit = { is_alive = yes }
			trigger_event = { id = WCDRA.1005 days = 1 }
		}
	}
	on_complete = {
		# Last event?
	}
	on_expire = {
	}
	
	on_participant_death = {
	
	}
	on_invalidated = {
	}
}
