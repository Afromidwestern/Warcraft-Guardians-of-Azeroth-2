set_agp_heir_interaction = {
	category = interaction_category_diplomacy
	common_interaction = yes
	icon = debug_take_title
	desc = set_agp_heir_interaction_desc

	auto_accept = yes
	use_diplomatic_range = no
	ignores_pending_interaction_block = yes
	
	is_shown = {
		scope:actor = {
			is_ai = no
			NOT = { this = scope:recipient }
			NOT = { exists = player_heir }
			OR = {
				has_government = republic_government
				has_government = theocracy_government
			}
			primary_title = { NOT = { has_title_law = agp_theocracy_elective_succession_law } }
			scope:recipient = {
				OR = {
					is_courtier_of = prev
					is_vassal_of = prev
				}
			}
			trigger_if = {
				limit = {
					has_variable = heir_agp
				}
				NOT = { var:heir_agp = scope:recipient }
			}
		}
	}
	
	on_accept = {
		scope:actor = {
			#настоящее назначение наследника
			set_variable = {
				name = heir_agp
				value = scope:recipient
			}
			
			#чтобы было видно в гуи, но set_designated_heir ни на что не влияет
			#set_designated_heir = scope:recipient
		}
	}
}
