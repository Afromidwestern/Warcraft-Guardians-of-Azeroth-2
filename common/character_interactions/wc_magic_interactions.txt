	category = interaction_category_wc_magic
	icon = order_arcane_intellect_interaction

	is_shown = {
		scope:actor = {
			is_magic_user_trigger = yes
		}
	}

	on_accept = {
		set_variable = {
			name = spellbook_open
			value = yes
		}
	}
	auto_accept = yes
}
