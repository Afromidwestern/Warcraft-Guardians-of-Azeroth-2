start_stasis_trap = {
	interface_priority = 70
	category = interaction_category_hostile

	send_name = START_SCHEME

	scheme = stasis_trap_scheme
	ignores_pending_interaction_block = yes
	
	is_shown = {
		scope:actor = { has_perk = stasis_trap } # Scheme unlocked by the Kidnapper Perk
		NOR = {
			scope:recipient = scope:actor
		}

		#exists = scope:recipient.court_owner
		
		NAND = { # Game Rules!
			has_game_rule = no_players_multiplayer_murder_schemes
			scope:actor = {
				is_ai = no
			}
			scope:recipient = {
				is_ai = no
			}
		}
		
		NAND = {
			has_game_rule = no_player_families_multiplayer_murder_schemes
			scope:actor = {
				is_ai = no
			}
			scope:recipient = {
				OR = {
					is_ai = no
					any_close_family_member = {
						AND = {
							is_ai = no
							NOT = { this = scope:actor }
						}
					}
				}
			}
		}
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			can_start_scheme = {
				type = stasis_trap_scheme
				target = scope:recipient
			}
		}
		scope:recipient = { NOT = { has_strong_hook = scope:actor } }

		custom_description = {
			text = stasis_trap_already_stunned

			scope:recipient = {
					NOT = { has_character_modifier = stasis_trap_stunned }
			} 
		}

		#Warcraft
		custom_description = {
			text = stasis_trap_you_are_stunned
			
			scope:actor = {
				NOT = { has_character_modifier = stasis_trap_stunned }
			} 
		}
	}

	desc = {
		triggered_desc = {
			trigger = {
				scope:actor = {
					can_start_scheme = {
						type = stasis_trap_scheme
						target = scope:recipient
					}
				}
			}
			desc = scheme_interaction_tt_stasis_trap_approved
		}
	}

	on_accept = {
		scope:actor = {
			stress_impact = {
				compassionate = medium_stress_impact_gain
				honest = minor_stress_impact_gain
				just = minor_stress_impact_gain
			}
		}
		hidden_effect = {
			scope:actor = {
				send_interface_toast = {
					title = start_stasis_trap_notification

					left_icon = scope:actor					
					right_icon = scope:recipient

					start_scheme = {
						type = stasis_trap_scheme
						target = scope:recipient
					}

					show_as_tooltip = {
						stress_impact = {
							compassionate = medium_stress_impact_gain
							honest = minor_stress_impact_gain
							just = minor_stress_impact_gain
						}
					}
				}
			}
		}
	}
	
	auto_accept = yes
}
