# Interation to start a plague on another character
start_plague_interaction = {
	category = interaction_category_hostile
	is_highlighted = {
		always = yes
	}
	icon = "gfx/interface/icons/character_interactions/declare_war_interaction.dds"

	send_name = start_plague_interaction

	interface_priority = 90
	use_diplomatic_range = yes

	desc = start_plague_interaction_desc

	greeting = negative
	notification_text = start_plague_interaction_notification

	is_shown = {
		scope:actor = {
			has_trait = lich_king
		}
	}
	is_valid_showing_failures_only = {
		scope:actor = {
			can_start_scheme = {
				type = plague
				target = scope:recipient
			}
		}
	}

	on_accept = {
		scope:actor = {
			trigger_event = plague_maintenance.100
		}
	}

	auto_accept = yes

	cost = {
		piety = 1000
	}

	ignores_pending_interaction_block = yes
	common_interaction = yes

	scheme = plague

	cooldown_against_recipient = {
		years = 20
	}
	cooldown = {
		years = 5
	}
}
