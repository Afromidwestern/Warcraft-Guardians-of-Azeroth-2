tradition_magic_wielders = {
	category = combat
	layers = {
		0 = martial
		1 = western
		4 = bow.dds
	}
	
	
	parameters = {
		unlock_maa_casters = yes
	}
	character_modifier = {
		casters_max_size_add = 2
		casters_damage_mult = 0.2
		casters_toughness_mult = 0.2
		casters_maintenance_mult = -0.3
		fort_level = 1
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			# Ethos requirement.
			if = {
				limit = {
					NOR = {
						# Stoic
						any_in_list = {
							list = traits
							this = flag:ethos_stoic
						}
						# Bureaucratic
						any_in_list = {
							list = traits
							this = flag:ethos_bureaucratic
						}
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_stoic_or_bureaucratic_desc
				}
			}
			# Certain number of archer MaA requirement.
			if = {
				limit = {
					scope:character = {
						number_maa_soldiers_of_base_type = {
							type = archers
							# Remember to sync the below with longbow_competitions_minimum_archer_maa_value.
							value < 1200
						}
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = culture_head_lacks_archer_maa_desc
				}
			}
			
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = {
				OR = {
					culture_has_massive_aerial_maa = yes
					culture_has_heavy_cavalry_maa = yes
					culture_has_massive_infantry_maa = yes
				}
			}
			multiply = 0
		}
	}
}
