# Feudal Raiding on Land
tradition_might_is_right = {
	category = societal

	layers = {
		0 = martial
		1 = mediterranean
		4 = fight.dds
	}
	
	is_shown = {
		heritage_gives_sea_raiding_trigger = no
		NOT = { scope:character.faith = { like_pacifism_faith_trigger = yes } }
	}
	
	can_pick = {
		custom_tooltip = {
			text = culture_not_pacifist_trigger_desc
			culture_not_pacifistic_trigger = yes
		}
		
		custom_tooltip = {
			text = cannot_have_tradition_isolationist_desc
			NOT = { any_in_list = { list = traits this = flag:tradition_isolationist } }
		}
		
		custom_tooltip = {
			text = cannot_have_tradition_practiced_pirates
			NOT = { any_in_list = { list = traits this = flag:tradition_practiced_pirates } }	
		}
	}
	
	parameters = {
		culture_can_raid_even_if_feudal = yes
		prowess_traits_more_valued = yes
	}
	county_modifier = {
		monthly_county_control_change_factor = -0.2
	}

	character_modifier = {
		knight_effectiveness_mult = 0.15
		opinion_of_liege = -10
		cultural_head_fascination_mult = -0.1
		ai_boldness = 25
	}
	
	cost = {
		prestige = {
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			# Ethos requirement.
			if = {
				limit = {
					NOR = {
						any_in_list = { list = traits this = flag:ethos_bellicose }
						any_in_list = { list = traits this = flag:ethos_stoic }
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_or_stoic_desc
				}
			}
			multiply = tradition_replacement_cost_if_relevant
		}
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = {
				NOR = { 
					any_in_list = { list = traits this = flag:ethos_bellicose }
					any_in_list = { list = traits this = flag:ethos_stoic }
				}
			}
			multiply = 0
		}
	}
}

#Mak'gora
tradition_makgora_trials_by_combat = {
	category = societal

	layers = {
		0 = intrigue
		1 = western
		4 = fight.dds
	}

	is_shown = {
	}

	can_pick = {
	}

	parameters = {
		has_access_to_makgora_trials_by_combat = yes
		blademaster_traits_give_learning = yes
		craven_trait_costs_prestige = yes
	}

	character_modifier = {
		knight_limit = 1
	}

	cost = {
		prestige = {
			# Base cost.
			add = {
				value = tradition_base_cost
				desc = BASE
				format = "BASE_VALUE_FORMAT"
			}
			# Ethos requirement.
			if = {
				limit = {
					NOR = {
						# Bellicose
						any_in_list = {
							list = traits
							this = flag:ethos_bellicose
						}
						# Stoic
						any_in_list = {
							list = traits
							this = flag:ethos_stoic
						}
						# Courtly
						any_in_list = {
							list = traits
							this = flag:ethos_courtly
						}
					}
				}
				add = {
					value = tradition_incompatible_ethos_penalty
					desc = not_bellicose_stoic_or_courtly_desc
				}
			}
			if = {
				limit = {
					trigger_if = {
						limit = {
							scope:character = {
								is_ai = no
							}
						}
						NOT = {
							any_ruler = {
								count >= 3
								culture = prev
								primary_title.tier >= tier_county
								has_trait = education_martial_4
								learning >= 10
							}
						}
					}
					trigger_else_if = {
						limit = {
							scope:character = {
								has_trait = education_martial_4
								learning >= 10
							}
						}
						always = no
					}
					trigger_else_if = {
						limit = {
							scope:character.culture = {
								OR = {
									like_heritage_brythonic_trigger = yes
									like_heritage_central_germanic_trigger = yes
									like_heritage_frankish_trigger = yes
									like_heritage_goidelic_trigger = yes
									like_heritage_iberian_trigger = yes
									like_heritage_latin_trigger = yes
									like_heritage_west_germanic_trigger = yes
								}
							}
						}
						always = no
					}
					trigger_else = {
						always = no
					}
				}
				add = {
					value = tradition_unfulfilled_criteria_penalty
					desc = rulers_with_martial_edu_and_learning_desc
				}
			}

			multiply = tradition_replacement_cost_if_relevant
		}
	}

	ai_will_do = { 
		value = 100 
		if = { #Ogres and orcs are much more likely to have this tradition
			limit = {
				OR = {
					has_trait = creature_orc
					has_trait = creature_ogre
				}
			}
			multiply = 2.5
		}
	}
}
