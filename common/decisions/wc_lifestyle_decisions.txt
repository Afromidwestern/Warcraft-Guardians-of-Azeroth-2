unleash_rage = {
	desc = unleash_rage_desc

	cooldown = { days = 182 }

	is_shown = {
		AND = {
			has_perk = unleashed_rage

			NOT = {
				has_character_modifier = unleashed_rage_modifier
			}
		}
	}

	effect = {
		add_character_modifier = {
			modifier = unleashed_rage_modifier
			days = 30
		}
	}
}

reign_in_rage = {
	desc = reign_in_rage_desc
	is_shown = {
		AND = {
			has_perk = unleashed_rage

			has_character_modifier = unleashed_rage_modifier
		}
	}

	effect = {
		remove_character_modifier = unleashed_rage_modifier
	}
}
