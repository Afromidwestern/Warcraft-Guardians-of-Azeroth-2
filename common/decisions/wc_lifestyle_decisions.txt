#Warcraft
unleash_rage = {
	desc = unleash_rage_desc
	is_shown = {
		AND = {
			has_perk = physical_tree_3_perk_2

			NOT = {
				has_character_modifier = unleashed_rage
			}
		}
	}

	is_valid = {
		NOT = {
			has_character_flag = unleashed_rage
		}
	}

	effect = {
		add_character_flag = {
			flag = unleashed_rage
			days = 182
		}

		add_character_modifier = {
			modifier = unleashed_rage
			days = 30
		}
	}
}

reign_in_rage = {
	desc = reign_in_rage_desc
	is_shown {
		AND = {
			has_perk = physical_tree_3_perk_2

			has_character_modifier = unleashed_rage
		}
	}

	effect = {
		remove_character_modifier = unleashed_rage
	}
}
