elemental_water_magic_tree_1_perk_1 = {
	lifestyle = elemental_water_magic_lifestyle
	tree = elemental_water_magic_tree_1
	position = { 2 0 }
	icon = node_diplomacy

	auto_selection_weight = {
		value = wc_magic_lifestyle_auto_selection_weight_base_value
		if = {
			limit = { MAGIC_magic_lifestyle_auto_selection_weight_add_base_trigger = { MAGIC = elemental_water } }
			add = wc_magic_lifestyle_auto_selection_weight_add_to_base_value
		}
		if = { # All 3 because there is only one tree
			limit = { has_focus_elemental_water_magic_trigger = yes }
			multiply = wc_magic_perk_auto_selection_weight_multiply_from_focus_value
		}
		if = {
			limit = { can_openly_learn_elemental_water_magic_trigger = no }
			multiply = 0
		}
	}

	effect = {
		wc_add_MAGIC_experience_effect = { MAGIC = elemental_water EXP = wc_magic_user_1_threshold_value } # Trait exp
	}
}
