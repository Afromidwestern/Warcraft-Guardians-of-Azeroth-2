physical_tree_1_perk_1 = {
	lifestyle = physical_lifestyle
	tree = stamina
	position = { 2 0 }
	icon = node_martial

	auto_selection_weight = {
		#Basically for this one I'm trying to make sure people who fit the requirements for the other two and are in somewhat 
		#bad shape pick this one
		value = wc_physical_lifestyle_auto_selection_weight_base_value

		if = { #Base
			limit = { health_tree_auto_selection_weight_add_base_trigger = yes }
			add = wc_physical_lifestyle_auto_selection_weight_add_to_base_value
		}

		if = { #Education
			limit = { has_trait = education_intrigue }
			add = wc_physical_lifestyle_auto_selection_weight_education_intrigue_add_to_base_value
		} 
		else_if = {
			limit = {
				AND = {
					OR = {
						has_trait = education_martial_prowess
						has_trait = education_martial
					}
					is_hunter_trigger = yes
				}
			}
			add = wc_physical_lifestyle_auto_selection_weight_education_martial_and_hunter_add_to_base_value
		} 
		else_if = {
			limit = {
				has_trait = education_martial
			}
			add = wc_physical_lifestyle_auto_selection_weight_education_martial_add_to_base_value
		}
		else_if = {
			limit = {
				has_trait = education_learning
			}
			add = wc_physical_lifestyle_auto_selection_weight_education_learning_add_to_base_value
		}

		if = {
			limit = { has_focus = physical_focus_1 }
			multiply = wc_magic_lifestyle_perk_auto_selection_weight_multiply_from_focus_value
		}
	}
}

			limit = { has_focus = physical_focus_3 }

		}
