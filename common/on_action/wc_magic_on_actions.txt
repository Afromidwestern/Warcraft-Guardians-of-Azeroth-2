cast_spell_on_action = {
	effect = {
		# which spell is in the list but tick is up
		every_in_list = {
			variable = current_spells_casting
			save_scope_as = current_spell
			if = {
				limit = {
					NOT = {
						has_character_flag = $SPELL$_remaining_cd
					}
				}
				set_variable = cast_spell_now
				cast_spell_effect = yes
			}
			remove_list_variable = {
				name = current_spells_casting
				target = scope:current_spell
			}
		}
		
	}
}
