titan_forge_cooldown_time_of_forge = {

	if = {
		limit = {
			save_temporary_scope_value_as = {
				name = max_level
				value = 1
			}
			save_temporary_scope_value_as = {
				name = cooldown_time
				value = 10000
			}
		}
	}
	every_directly_owned_province = {
		if = {
			limit = { 
				has_building = titan_forge_02 
				scope:max_level < 2
				save_temporary_scope_value_as = {
					name = max_level
					value = 2
				}
				save_temporary_scope_value_as = {
					name = cooldown_time
					value = titan_forge_02_cooldown_time
				}
			}
		}
		else if = {
			limit = { 
				has_building = titan_forge_03 
				scope:max_level < 3

					name = max_level
					value = 3
				}
				save_temporary_scope_value_as = {
					name = cooldown_time
					value = titan_forge_03_cooldown_time
				}
			}
		}
		else if = {
			limit = { 
				has_building = titan_forge_04 
				scope:max_level < 4
				save_temporary_scope_value_as = {
					name = max_level
					value = 4
				}
				save_temporary_scope_value_as = {
					name = cooldown_time
					value = titan_forge_04_cooldown_time
				}
			}
		}
		else if = {
			limit = { 
				has_building = titan_forge_05 
				scope:max_level < 5
				save_temporary_scope_value_as = {
					name = max_level
					value = 5
				}
				save_temporary_scope_value_as = {
					name = cooldown_time
					value = titan_forge_05_cooldown_time
