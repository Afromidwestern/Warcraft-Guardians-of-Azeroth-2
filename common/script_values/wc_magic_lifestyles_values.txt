wc_light_magic_auto_selection_value = {
	# Addings
	if = { # Base
		limit = {
			MAGIC_magic_lifestyle_auto_selection_weight_add_base_trigger = { MAGIC = light }
		}
		add = wc_magic_lifestyle_auto_selection_weight_add_to_base_value
	}
	
	if = { # Education
		limit = { has_trait = education_learning }
		add = wc_magic_lifestyle_auto_selection_weight_education_learning_add_to_base_value
	}
	else_if = {
		limit = { has_trait = education_martial }
		add = wc_magic_lifestyle_auto_selection_weight_education_martial_add_to_base_value
	}
	
	## Multipliers
	# Common
	multiply = wc_magic_lifestyle_focus_selection_weight_multiply_all_magic_value
	# Light Magic
	if = {
		limit = {
			magic_is_shunned_or_illegal_trigger = { magic = light }
			NOT = { has_trait = cynical }
		}
		multiply = 0
	}
}
