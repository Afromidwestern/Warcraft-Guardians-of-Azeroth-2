culture_ai_weight_caster = {
	value = 168.7
	
	if = { # Cultures that favours casters
		limit = {
			OR = {
				culture_group = culture_group:elf_group
				culture = culture:dalaranian
			}
		}
		add = 40
	}
	if = { # Cultures that dislike casters
		limit = {
				taur_culture_trigger = yes
		}
		add = -60
	}
}

	value = -4

	}
}
