# Sacrifice Value

sacrifice_change_old_god_power_value = {
	if = {
		limit = { scope:victim = { highest_held_title_tier >= tier_empire } }
		add = 500
	}
	else_if = {
		limit = { scope:victim = { highest_held_title_tier >= tier_kingdom } }
		add = 400
	}
	else_if = {
		limit = { scope:victim = { highest_held_title_tier >= tier_duchy } }
		add = 300
	}
	else_if = {
		limit = { scope:victim = { is_ruler = yes } }
		add = 100
	}
	else_if = {
		limit = { scope:victim = { any_close_family_member = { highest_held_title_tier >= tier_empire } } }
		add = 300
	}
	else_if = {
		limit = { scope:victim = { any_close_family_member = { highest_held_title_tier >= tier_kingdom } } }
		add = 200
	}
	else_if = {
		limit = { scope:victim = { any_close_family_member = { highest_held_title_tier >= tier_duchy } } }
		add = 100
	}
	else_if = {
		limit = { scope:victim = { OR = { is_lowborn = no any_close_family_member = { is_ruler = yes } } } }
		add = 50
	}
	else = { add = 25 }
	if = {
		limit = { scope:victim = { is_close_family_of = scope:executioner } }
		multiply = 1.5
	}
	floor = yes
}
