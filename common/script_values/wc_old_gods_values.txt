# Sacrifice Value

sacrifice_change_old_god_power_value = {
	if = {
		limit = {
			scope:victim = {
				OR = {
					highest_held_title_tier >= tier_empire
					any_close_family_member = { highest_held_title_tier >= tier_empire }
				}
			}
			add = 400
		}
	}
	else_if = {
		limit = {
			scope:victim = {
				OR = {
					highest_held_title_tier >= tier_kingdom
					any_close_family_member = { highest_held_title_tier >= tier_kingdom }
				}
			}
			add = 200
		}
	}
	else_if = {
		limit = {
			scope:victim = {
				OR = {
					highest_held_title_tier >= tier_duchy
					any_close_family_member = { highest_held_title_tier >= tier_duchy }
				}
			}
			add = 100
		}
	}
	else_if = {
		limit = {
			scope:victim = {
				OR = {
					is_ruler = yes
					is_lowborn = no
					any_close_family_member = { is_ruler = yes }
				}
			}
			add = 50
		}
	}
	else = { add = 25 }
	# floor = yes
}
