		limit = {
			OR = {
				has_trait = wc_lifestyle_elemental_fire_magic
				has_trait = wc_lifestyle_elemental_fire_spirit_magic
				has_trait = wc_lifestyle_elemental_fire_decay_magic

			value = elemental_fire_magic_lifestyle_perks

		limit = { has_trait = magic_good_2 }
