# Weight used in selecting targets for the automated Horde invasions
horde_invasion_target_character_weight = {
	value = 0.001
	
	if = {
		limit = {
			OR = {
				this = title:k_stormwind
				this = title:k_blackrock
				this = title:k_ironforge
				this = title:k_wildhammer
			}
		}
		add = 10000
	}
	else_if = {
		limit = {
			title_province = { geographical_region = special_horde_conquest_region_prio_1 }
		}
		add = 1000
	}
	else_if = {
		limit = {
			title_province = { geographical_region = special_horde_conquest_region_prio_2 }
		}
		add = 100
	}
}
