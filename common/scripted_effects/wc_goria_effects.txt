create_gorian_empire_scripted_effect = {
	
	spawn_gorian_troops_effect = yes

	unlock_innovation = innovation_gorian_chariot
	
	custom_tooltip = create_gorian_empire
	show_as_tooltip = {
		every_held_title = {
			custom = create_title_every_held_kingdom
			limit = {
				tier = tier_kingdom
			}
			custom_tooltip = create_gorian_empire_de_jure_changes
		}
	}

	hidden_effect = {
		save_scope_as = founder
		primary_title = {
			save_scope_as = old_title
		}

		create_dynamic_title = {
			tier = empire
			name = NEW_CREATED_TITLE_NAME
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}

		scope:new_title = {
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}

		resolve_title_and_vassal_change = scope:change

  		every_held_title = {
			limit = {
				tier = tier_kingdom
			}
			if = {
				limit = {
					#Check if you need to notify a player
					exists = empire
					empire = {
						exists = holder
						holder = {
							NOT = { this = root }
							is_ai = no
						}
					}
				}
				add_to_temporary_list = kingdoms_for_notification
				root = {
					save_temporary_scope_value_as = {
						name = send_notifications
						value = yes
					}
				}
			}
			set_de_jure_liege_title = scope:new_title
		}

		every_sub_realm_county = {
			limit = {
				exists = kingdom
				NOT = { exists = kingdom.holder }
				holder.top_liege = root
				kingdom = {
					save_temporary_scope_as = test_kingdom
				}
				holder.top_liege = {
					completely_controls = scope:test_kingdom
				}
			}
			if = {
				limit = {
					NOT = {
						kingdom = {
							is_in_list = additional_de_jure_kingdoms
						}
					}
				}
				kingdom = {
					set_de_jure_liege_title = scope:new_title
					add_to_list = additional_de_jure_kingdoms
				}
			}
		}

		scope:new_title = {
			set_coa = scope:old_title
			set_color_from_title = scope:old_title
			set_capital_county = scope:old_title.title_capital_county
		}
		set_primary_title_to = scope:new_title

		trigger_event = gorian_restoration.0001

		every_player = {
			if = {
				limit = {
					top_liege = scope:founder
					NOT = { this = root }
				}
				trigger_event = major_decisions.1104
			}
			else_if = {
				#Notify any players who lose de jure territory
				limit = {
					exists = scope:send_notifications
					NOT = { this = root }
					NOT = { top_liege = scope:founder }
					any_held_title = {
						any_in_de_jure_hierarchy = {
							continue = {
								tier > tier_kingdom
							}
							tier = tier_kingdom
							is_in_list = kingdoms_for_notification
						}
					}
				}
				every_held_title = {
					every_in_de_jure_hierarchy = {
						continue = {
							tier > tier_kingdom
						}
						limit = {
							tier = tier_kingdom
							is_in_list = kingdoms_for_notification
						}
						add_to_list = notification_titles
					}
				}
				if = {
					limit = {
						any_in_list = {
							list = notification_titles
							count > 0
						}
					}
					trigger_event = major_decisions.1105
				}
			}
		}
	}	
}

spawn_gorian_troops_effect = {
	if = {
		limit = { exists = capital_province }
		spawn_army = {
			uses_supply = no
			inheritable = yes
			name = scourge_event_troops
			men_at_arms = {
				type = mauler
				stacks = 6
			}
			men_at_arms = {
				type = magi
				stacks = 6
			}
			men_at_arms = {
				type = gorian_chariot
				stacks = 4
			}
			men_at_arms = {
				type = trebuchet
				stacks = 1
			}
			location = capital_province
		}
	}
}
