get_pol_faction_variables_effect = {
	remove_variable = temp_leader
	clear_variable_list = temp_members
	remove_variable = temp_strength
	
	set_variable = {
		name = temp_strength
		value = 0
	}
	
	global_var:$faction$_leader = {
		root = {
			set_variable = {
				name = temp_leader
				value = prev
			}
		}
	}
	every_in_global_list = {
		variable = $faction$_members
		if = {
			limit = { NOT = { this = global_var:$faction$_leader } }
			root = {
				add_to_variable_list = {
					name = temp_members
					target = prev
				}
			}
		}
		root = {
			change_variable = {
				name = temp_strength
				add = prev.max_military_strength
			}
		}
	}
}
