purge_historical_character_effect = {
	set_to_lowborn = yes
	# Basically reset parents
	if = {
		limit = { exists = father }
		set_father = dummy_male
	}
	if = {
		limit = { exists = real_father }
		set_real_father = dummy_male
	}
	if = {
		limit = { exists = mother }
		set_real_father = dummy_female
	}
	death = { death_reason = death_vanished }
}
