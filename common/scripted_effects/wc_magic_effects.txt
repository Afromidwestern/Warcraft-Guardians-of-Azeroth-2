add_collected_souls_effect = {
	if = {
		limit = {
			NOT = {
				has_variable = collected_souls
			}
		}
		set_variable = {
			name = collected_souls
			value = 0
		}
	}
	
	change_variable = {
		name = collected_souls
		add = $AMOUNT$
	}
	
	if = {
		limit = {
			var:collected_souls < 0
		}
		
		set_variable = {
			name = collected_souls
			value = 0
		}
	}
	
	if = {
		limit = {
			$AMOUNT$ = 1
		}
		custom_description = {
			text = ADD_SOUL
			value = $AMOUNT$
		}
	}
	else_if = {
		limit = {
			$AMOUNT$ > 0
		}
		custom_description = {
			text = ADD_SOULS
			value = $AMOUNT$
		}
	}
	else_if = {
		limit = {
			$AMOUNT$ = -1
		}
		custom_description = {
			text = REMOVE_SOUL
			value = $AMOUNT$
		}
	}
	else_if = {
		limit = {
			$AMOUNT$ < 0
		}
		custom_description = {
			text = REMOVE_SOULS
			value = $AMOUNT$
		}
	}
}
