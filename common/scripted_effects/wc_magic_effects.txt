add_collected_souls_effect = {
	if = {
		limit = {
			NOT = {
				has_variable = collected_souls
			}
		}
		set_variable = {
			name = collected_souls
			value = 0
		}
	}
	
	change_variable = {
		name = collected_souls
		add = $AMOUNT$
	}
	
	if = {
		limit = {
			var:collected_souls < 0
		}
		
		set_variable = {
			name = collected_souls
			value = 0
		}
	}
}
