wc_regenerate_mana_effect = {
	if = {
		limit = {
			has_variable = wc_current_mana_variable
			var:wc_current_mana_variable < wc_max_mana_value
		}
		change_variable = { name = wc_current_mana_variable add = 1 }
	}
	else = { set_variable = { name = wc_current_mana_variable value = wc_max_mana_value } }
}
