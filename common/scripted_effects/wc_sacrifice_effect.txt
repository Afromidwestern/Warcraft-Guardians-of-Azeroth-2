wc_sacrifice_character_effect = {
	$SACRIFICER$ = { save_scope_as = sacrificer }
	$OFFERING$ = { save_scope_as = offering }

	scope:sacrificer = {
		custom_tooltip = wc_sacrifice_character_effect_tooltip
		hidden_effect = {
			if = {
				limit = { faith = { has_doctrine_parameter = old_god_worship_parameter } }

				faith = {
					switch = {
						trigger = has_doctrine
						doctrine_old_god_cthun = { change_global_variable = { name = global_cthun_power_variable add = sacrifice_change_old_god_power_value } }
						doctrine_old_god_yoggsaron = { change_global_variable = { name = global_yoggsaron_power_variable add = sacrifice_change_old_god_power_value } }
						doctrine_old_god_nzoth = { change_global_variable = { name = global_nzoth_power_variable add = sacrifice_change_old_god_power_value } }
						doctrine_old_god_yshaarj = { change_global_variable = { name = global_yshaarj_power_variable add = sacrifice_change_old_god_power_value } }
						doctrine_old_god_ghuun = { change_global_variable = { name = global_ghuun_power_variable add = sacrifice_change_old_god_power_value } }
					}
				}

				if = {
					limit = { has_variable = wc_divine_grace_variable }
					change_variable = { name = wc_divine_grace_variable add = sacrifice_change_old_god_power_value }
				}
				else = { set_variable = { name = wc_divine_grace_variable value = sacrifice_change_old_god_power_value } }
			}
		}
	}
	custom_tooltip = wc_sacrifice_divine_effect_tooltip
}
