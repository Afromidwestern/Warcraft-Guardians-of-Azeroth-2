get_alliance_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		remove_variable = temp_leader
		clear_variable_list = temp_members
		set_variable = {
			name = temp_strength
			value = 0
		}
		
		global_var:alliance_leader = {
			root = {
				set_variable = {
					name = temp_leader
					value = prev
				}
			}
		}
		every_in_global_list = {
			variable = alliance_members
			if = {
				limit = { NOT = { this = global_var:alliance_leader } }
				root = {
					add_to_variable_list = {
						name = temp_members
						target = prev
					}
					change_variable = {
						name = temp_strength
						add = prev.max_military_strength
					}
				}
			}
		}
	}
}

clear_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		remove_variable = temp_leader
		clear_variable_list = temp_members
	}
}
