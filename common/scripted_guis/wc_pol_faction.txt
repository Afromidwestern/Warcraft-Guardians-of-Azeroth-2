get_alliance_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		get_pol_faction_variables_effect = { faction = alliance }
	}
}
get_horde_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		get_pol_faction_variables_effect = { faction = horde }
	}
}

clear_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		remove_variable = temp_leader
		clear_variable_list = temp_members
		remove_variable = temp_strength
	}
}
