get_alliance_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		get_pol_faction_variables_effect = { faction = alliance }
	}
}
get_horde_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		get_pol_faction_variables_effect = { faction = horde }
	}
}

clear_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		remove_variable = temp_leader
		clear_variable_list = temp_members
		remove_variable = temp_strength
	}
}

pol_factions_toggle = {
	scope = character
	
	is_shown = {
		has_variable = pol_factions_toggle
	}

	effect = {
		if = {
			limit = {
				has_variable = pol_factions_toggle
			}
			remove_variable = pol_factions_toggle
		}
		else = {
			set_variable = pol_factions_toggle
		}
	}
}

pol_factions_close = {
	scope = character
	
	is_shown = {
		always = yes
	}

	effect = {
		remove_variable = pol_factions_toggle
		random_living_character = {
			limit = { has_variable = leader_of_your_society }
			remove_variable = leader_of_your_society
		}
	}
}
