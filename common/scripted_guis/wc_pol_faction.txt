get_alliance_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		get_pol_faction_variables_effect = { faction = alliance }
	}
}
get_horde_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		get_pol_faction_variables_effect = { faction = horde }
	}
}

clear_pol_faction = {
	scope = character
	is_shown = {
		always = yes
	}
	is_valid = {
		always = yes
	}
	effect = {
		clear_pol_faction_variables_effect = yes
	}
}

pol_factions_toggle = {
	scope = character
	
	is_shown = {
		has_variable = pol_factions_toggle
	}

	effect = {
		if = {
			limit = {
				has_variable = pol_factions_toggle
			}
			remove_variable = pol_factions_toggle
		}
		else = {
			set_variable = pol_factions_toggle
		}
	}
}

pol_factions_close = {
	scope = character
	
	is_shown = {
		always = yes
	}

	effect = {
		remove_variable = pol_factions_toggle
	}
}
