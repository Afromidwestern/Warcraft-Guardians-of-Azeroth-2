clean_shaven_modifier = {
	# The chance is 30% + This Modifier
	modifier = {
		save_temporary_scope_value_as = {
			name = negative
			value = flag:$negative$
		}
		
		add = {
			value = 0
			if = {
				limit = { portrait_northern_clothing_trigger = yes }
				add = -5
			}
			if = {
				limit = { portrait_steppe_clothing_trigger = yes }
				add = -5
			}
			if = {
				limit = { portrait_mena_clothing_trigger = yes }
				add = -25
			}
			
			# Beard cultures
			# Elves are 50% clean shaven
			if = {
				limit = { portrait_high_elven_beards_trigger = yes }
				add = 20
			}
			# Ogres are 80% clean shaven (should go before orcs since share the same beards with them)
			else_if = {
				limit = { portrait_ogre_beards_trigger = yes }
				add = 50
			}
			# Orcs are 35% clean shaven
			else_if = {
				limit = { portrait_orcish_beards_trigger = yes }
				add = 5
			}
			#Dwarves never clean shaven
			else_if = {
				limit = { portrait_dwarven_beards_trigger = yes }
				add = -30
			}
			#Vrykul are 5% clean shaven
			else_if = {
				limit = { portrait_vrykul_beards_trigger = yes }
				add = -25
			}
			
			# Religions
			#Muslim-like are never clean shaven
			if = {
				limit = { like_islam_religion_trigger = yes }
				add = -30
			}
			
			#Must be at the end
			if = {
				limit = {
					scope:negative = flag:yes
				}
				multiply = -1
			}
			
			min = 0
			max = 70
		}
	}
}
bald_modifier = {
	# The chance is 0% + This Modifier
	modifier = {
		save_temporary_scope_value_as = {
			name = negative
			value = flag:$negative$
		}
		add = {
			value = 0
			
			#Hair cultures
			# Dwarves are 40% bald
			if = {
				limit = { portrait_dwarven_hairstyles_trigger = yes }
				add = 40
			}
			# Orcs are 40% bald
			else_if = {
				limit = { portrait_orcish_hairstyles_trigger = yes }
				add = 40
			}
			# Ogres are 80% bald
			else_if = {
				limit = { portrait_ogre_hairstyles_trigger = yes }
				add = 80
			}
			# Blood trolls are 70% bald
			else_if = {
				limit = { portrait_blood_troll_hairstyles_trigger = yes }
				add = 70
			}
			# Trolls are 40% bald
			else_if = {
				limit = { portrait_troll_hairstyles_trigger = yes }
				add = 40
			}
			
			if = {
				limit = {
					has_character_modifier = hajj_halq_modifier
				}
				add = 100
			}
			if = {
				limit = { scope:culture = { has_graphical_culture = no_hair_gfx } }
				add = 100
			}
			
			#Must be at the end
			if = {
				limit = {
					scope:negative = flag:yes
				}
				multiply = -1
			}
			
			if = {
				limit = { no_bald_trigger = yes }
				multiply = 0
			}
			
			min = 0
			max = 100
		}
	}
}
