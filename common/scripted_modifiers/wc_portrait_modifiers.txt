clean_shaven_modifier = {
	# The chance is 30% + This Modifier
	modifier = {
		save_temporary_scope_value_as = {
			name = negative
			value = flag:$negative$
		}
		
		add = {
			value = 0
			if = {
				limit = { portrait_northern_clothing_trigger = yes }
				add = -5
			}
			if = {
				limit = { portrait_steppe_clothing_trigger = yes }
				add = -5
			}
			if = {
				limit = { portrait_mena_clothing_trigger = yes }
				add = -25
			}
			# Elves are 50% clean shaven
			if = {
				limit = { portrait_high_elven_beards_trigger = yes }
				add = 20
			}
			# Orcs are 35% clean shaven
			if = {
				limit = { portrait_orcish_beards_trigger = yes }
				add = 5
			}
			# Ogres are 80% clean shaven
			if = {
				limit = { portrait_ogre_clothing_trigger = yes }
				add = 50
			}
			#Dwarves never clean shaven
			if = {
				limit = { portrait_dwarven_beards_trigger = yes }
				add = -30
			}
			#Vrykul are 5% clean shaven
			if = {
				limit = { portrait_vrykul_beards_trigger = yes }
				add = -25
			}
			
			#Must be at the end
			if = {
				limit = {
					scope:negative = flag:yes
				}
				multiply = -1
			}
		}
	}
}
bald_modifier = {
	# The chance is 0% + This Modifier
	modifier = {
		save_temporary_scope_value_as = {
			name = negative
			value = flag:$negative$
		}
		add = {
			value = 0
			# Dwarves are 40% bald
			if = {
				limit = {
					OR = {
						has_trait = creature_dwarf
						culture = { has_graphical_culture = dwarf_clothing_gfx }
					}
				}
				add = 40
			}
			# Orcs are 40% bald
			if = {
				limit = {
					OR = {
						has_trait = creature_orc
						culture = { has_graphical_culture = orcish_clothing_gfx }
					}
				}
				add = 40
			}
			# Ogres are 80% bald
			if = {
				limit = {
					OR = {
						has_trait = creature_ogre
						culture = { has_graphical_culture = ogre_portrait_gfx }
					}
				}
				add = 80
			}
			# Trolls are 40% bald
			if = {
				limit = {
					OR = {
						has_trait = creature_troll
						culture = { has_graphical_culture = trollish_clothing_gfx }
					}
				}
				add = 40
			}
			# Blood trolls are 70% bald (the modifier above also applies to them)
			if = {
				limit = {
					culture = { has_graphical_culture = blood_trollish_clothing_gfx }
				}
				add = 30
			}
			if = {
				limit = {
					has_character_modifier = hajj_halq_modifier
				}
				add = 100
			}
			if = {
				limit = { scope:culture = { has_graphical_culture = no_hair_gfx } }
				add = 100
			}
			
			#Must be at the end
			if = {
				limit = {
					scope:negative = flag:yes
				}
				multiply = -1
			}
			
			if = {
				limit = { no_bald_trigger = yes }
				multiply = 0
			}
		}
	}
}
