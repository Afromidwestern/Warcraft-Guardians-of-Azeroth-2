clean_shaven_modifier = {
	# The chance is 30% + This Modifier
	modifier = {
		save_temporary_scope_value_as = {
			name = negative
			value = flag:$negative$
		}
		
		add = {
			value = 0
			if = {
				limit = { portrait_northern_clothing_trigger = yes }
				add = -5
			}
			if = {
				limit = { portrait_steppe_clothing_trigger = yes }
				add = -5
			}
			if = {
				limit = { portrait_mena_clothing_trigger = yes }
				add = -25
			}
			# Elves are 50% clean shaven
			if = {
				limit = {
					OR = {
						is_elf_trigger = yes
						culture = { has_graphical_culture = high_elven_clothing_gfx }
					}
				}
				add = 20
			}
			# Orcs are 35% clean shaven
			if = {
				limit = {
					OR = {
						has_trait = creature_orc
						culture = { has_graphical_culture = orcish_clothing_gfx }
					}
				}
				add = 5
			}
			# Ogres are 80% clean shaven
			if = {
				limit = {
					OR = {
						has_trait = creature_ogre
						culture = { has_graphical_culture = ogre_portrait_gfx }
					}
				}
				add = 50
			}
			#Dwarves never clean shaven
			if = {
				limit = {
					OR = {
						has_trait = creature_dwarf
						culture = { has_graphical_culture = dwarf_clothing_gfx }
					}
				}
				add = -30
			}
			#Vrykul are 5% clean shaven
			if = {
				limit = {
					OR = {
						is_vrykul_trigger  = yes
						culture = { has_graphical_culture = vrykul_portrait_gfx }
					}
				}
				add = -25
			}
			
			#Must be at the end
			if = {
				limit = {
					scope:negative = flag:yes
				}
				multiply = -1
			}
		}
	}
}
bald_modifier = {
	# The chance is 0% + This Modifier
	modifier = {
		save_temporary_scope_value_as = {
			name = negative
			value = flag:$negative$
		}
		add = {
			value = 0
			# Dwarves are 40% bald
			if = {
				limit = {
					OR = {
						has_trait = creature_dwarf
						culture = { has_graphical_culture = dwarf_clothing_gfx }
					}
				}
				add = 40
			}
			# Orcs are 40% bald
			if = {
				limit = {
					OR = {
						has_trait = creature_orc
						culture = { has_graphical_culture = orcish_clothing_gfx }
					}
				}
				add = 40
			}
			# Ogres are 80% bald
			if = {
				limit = {
					OR = {
						has_trait = creature_ogre
						culture = { has_graphical_culture = ogre_portrait_gfx }
					}
				}
				add = 80
			}
			# Trolls are 40% bald
			if = {
				limit = {
					OR = {
						has_trait = creature_troll
						culture = { has_graphical_culture = trollish_clothing_gfx }
					}
				}
				add = 40
			}
			# Blood trolls are 70% bald (the modifier above also applies to them)

				}
				add = 30
			}
			if = {
				limit = {

			#Must be at the end
			if = {
				limit = {
					scope:negative = flag:yes
				}
				multiply = -1
			}
		}
	}
}
