attracted_to_race_modifier = {
	modifier = {
		add = {
			value = $value$
			# Humans find high elves attractive so they should tolerate other elves
			if = {
				limit = {
					$first$ = {
						OR = {
							is_elf_trigger = yes
							has_trait = creature_human
						}
					}
					$second$ = {
						OR = {
							is_elf_trigger = yes
							has_trait = creature_human
						}
					}
				}
				multiply = 0.5
			}
			#There are many examples in the lore
			if = {
				limit = {
					$first$ = {
						OR = {
							has_trait = creature_goblin
							has_trait = creature_gnome
						}
					}
					$second$ = {
						OR = {
							has_trait = creature_goblin
							has_trait = creature_gnome
						}
					}
				}
				multiply = 0.5
			}
		}
		$first$ = { NOT = { is_attracted_to_race_trigger = { CHARACTER = $second$ } } }
		desc = NOT_ATTRACTIVE_RACE
	}
}
