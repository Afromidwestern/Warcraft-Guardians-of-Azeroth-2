#Controls the can_raid rule.
can_raid_trigger = {
	OR = {
		government_has_flag = government_can_raid_rule
		faith = { has_doctrine_parameter = faith_can_raid }
		AND = {
			exists = dynasty
			dynasty = { has_dynasty_modifier = fp1_legacy_of_piracy_modifier }
		}
		trigger_if = {
			limit = {
				OR = {
					has_culture = culture:kvaldir
					has_culture = culture:saurok
					has_culture = culture:black_dragon
					has_culture = culture:northsea
					has_culture = culture:blackwater
					has_culture = culture:pirate
					has_culture = culture:gilblin
				}
			}
			scope:culture = { has_cultural_parameter = culture_can_raid_at_sea_even_if_feudal }
		}
		trigger_else = {
			scope:culture = { 
				has_cultural_parameter = culture_can_raid_even_if_feudal 
				has_government = feudal_government
			}
		}
	}
	NOT = {
		faith = {
			has_doctrine_parameter = holy_wars_forbidden
		}
	}
}
