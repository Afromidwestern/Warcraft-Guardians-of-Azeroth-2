		trigger_if = {

			limit = { exists = title:e_scourge.holder }
			character_is_on_same_continent_trigger = { CHARACTER = title:e_scourge.holder }
		}
		trigger_if = {
			limit = { exists = title:e_burning_legion.holder }
			character_is_on_same_continent_trigger = { CHARACTER = title:e_burning_legion.holder }
		}

			limit = { exists = title:e_black_empire.holder }

		}

	}
