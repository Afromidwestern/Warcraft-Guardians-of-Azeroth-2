like_north_germanic_group_trigger = {
	trigger_if = {
		limit = { exists = culture }
		culture = { save_temporary_scope_as = culture }
	}
	trigger_else = {
		save_temporary_scope_as = culture
	}
	
	scope:culture = { has_cultural_pillar = heritage_vrykulic }
}
