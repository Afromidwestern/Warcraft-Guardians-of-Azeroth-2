scandi_adventurer_region_trigger = {
	geographical_region = world_eastern_kingdoms
}
potential_western_scandi_adventurer_trigger = {
	# Warcraft
	like_north_germanic_group_trigger = yes
	NOT = { geographical_region = dlc_fp1_region_eastern_scandinavia_pool }
}
potential_eastern_scandi_adventurer_trigger = {
	# Warcraft
	like_north_germanic_group_trigger = yes
	geographical_region = dlc_fp1_region_eastern_scandinavia_pool
}

is_valid_target_for_western_scandi_adventurer_trigger = {
	NOR = {
		any_county_in_region = {
			region = $SCANDI_ADV_TARGET_REGION$
			significant_north_germanic_presence_trigger = { SCANDI_ADV_TARGET_REGION = $SCANDI_ADV_TARGET_REGION$ }
		}
		any_county_in_region = {
			region = $SCANDI_ADV_TARGET_REGION$
			count >= 3
			scattered_north_germanic_presence_trigger = yes
		}
	}
	any_county_in_region = {
		region = $SCANDI_ADV_TARGET_REGION$
		valid_adventurer_target_trigger = yes
	}
}
