### Frostwolf Exodus Triggers

# Frostwolf Exodus: These conditions must be met to continue event chain
frostwolf_exodus_d_frostwolf_trigger = {
	scope:frostwolf_holder = {
		has_primary_title = title:d_frostwolf
		is_alive = yes	
		is_imprisoned = no

		has_faith = faith:orcish_shamanism
	}
}

frostwolf_exodus_d_stormreaver_trigger = {
	scope:stormreaver_holder = {
		has_primary_title = title:d_stormreaver
		is_alive = yes	
		is_imprisoned = no

		has_faith = faith:orcish_fel
	}
}
