can_be_impotent_trigger = {
	can_have_children = yes
	NOT = { has_trait = impotent }
	
	is_dragon_race_trigger = no
	
	OR = {
		trigger_if = {
			limit = { is_female = yes }
			age >= female_fertility_age_value
		}
		trigger_else = {
			age >= male_fertility_age_value
		}
		
		trigger_if = {
			limit = { racial_age_multiplier_value > 1 }
			trigger_if = {
				limit = { is_female = yes }
				any_child = { count >= 5 }
			}
			trigger_else = {
				any_child = { count >= 7 }
			}
		}
	}
}
has_dire_or_giant_trigger = {
	OR = {
		has_trait = giant
		has_trait = dire
	}
}
dire_possible_trigger = {
	is_troll_trigger = yes
	NOR = {
		has_trait = giant
		has_trait = dwarf
	}
}
