can_be_impotent_trigger = {
	can_have_children = yes
	NOT = { has_trait = impotent }
	
	is_dragon_race_trigger = no
	
	OR = {
		trigger_if = {
			limit = { is_female = yes }
			age >= female_fertility_age_value
		}
		trigger_else = {
			age >= male_fertility_age_value
		}
		
		trigger_if = {
			limit = { racial_age_multiplier_value > 1 }
			trigger_if = {
				limit = { is_female = yes }
				any_child = { count >= 3 }
			}
			trigger_else = {
				any_child = { count >= 4 }
			}
		}
	}
}
