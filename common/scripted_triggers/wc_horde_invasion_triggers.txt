	is_alive = yes	
	is_imprisoned = no

	faith = faith:orcish_shamanism

	can_resettle_trigger = yes
	has_primary_title = { title = d_frostwolf }
}

	is_holy_order = no
	OR = {
		is_ruler = no
		is_landed = yes
	}
}
