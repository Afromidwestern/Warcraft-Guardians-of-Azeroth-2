realm_law_use_tribal_authority = {
	is_ruler = yes
	has_government = tribal_government
}
realm_law_use_dark_authority = {
	is_ruler = yes
	has_dark_government_trigger = yes
}

has_realm_crown_authority_trigger = {
	OR = {
		has_realm_law = crown_authority_0
		has_realm_law = crown_authority_1
		has_realm_law = crown_authority_2
		has_realm_law = crown_authority_3
	}
}
has_realm_tribal_authority_trigger = {
	OR = {
		has_realm_law = tribal_authority_0
		has_realm_law = tribal_authority_1
		has_realm_law = tribal_authority_2
		has_realm_law = tribal_authority_3
	}
}
has_realm_dark_authority_trigger = {
	OR = {
		has_realm_law = dark_authority_0
		has_realm_law = dark_authority_1
		has_realm_law = dark_authority_2
		has_realm_law = dark_authority_3
	}
}

has_monarchy_authority_level_trigger = {
	OR = {
		has_realm_law = crown_authority_$LEVEL$
		has_realm_law = dark_authority_$LEVEL$
	}
}
