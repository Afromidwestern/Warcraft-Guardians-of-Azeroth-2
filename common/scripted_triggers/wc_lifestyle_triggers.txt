can_openly_learn_light_magic_trigger = {
	trigger_if = {
		limit = {
			faith = { has_doctrine_parameter = elunic_light_magic_female_only }
			is_male = yes
		}
		is_cenarius_offspring_trigger = yes
	}
	faith = { has_doctrine_parameter = light_magic_openly_learnable }
}
