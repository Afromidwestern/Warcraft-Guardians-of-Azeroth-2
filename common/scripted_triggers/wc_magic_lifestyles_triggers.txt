MAGIC_magic_lifestyle_auto_selection_weight_add_base_trigger = {
	trigger_if = {
		limit = { faith = { has_doctrine_parameter = $MAGIC$_magic_approved } }

		OR = {
			has_trait = zealous
			is_clergy = yes
			has_trait = magic_good
			has_trait = magic_good_noninheritable
		}
	}
	trigger_else = { always = no }
}
