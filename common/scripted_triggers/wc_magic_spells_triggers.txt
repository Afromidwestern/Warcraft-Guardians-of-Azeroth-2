can_learn_spell_trigger = {
	OR = {
		AND = {
			has_variable_list = known_$CLASS$_spells
			any_in_list = { variable = known_$CLASS$_spells this = flag:$SPELL_NAME$ }
		}
		NOT = {	has_variable_list = known_$CLASS$_spells }
	}
}

has_unlocked_spell_rank = {
	OR = {
		AND = {
			has_variable_list = known_rank_$RANK_VALUE$_spells
			any_in_list = { variable = known_rank)$RANK_VALUE$_spells this = flag:$SPELL_NAME$ }
		}
		#specific spells will all ranks here
		flag:$SPELL_NAME$ = flag:gay_ass_spell
	}
}

can_cast_spell_trigger = {
	NOT = { target_has_max_stacks_trigger = { SPELL = $SPELL_NAME$ TARGET = $TARGET$ } }
	NOT = { has_variable = $SPELL_NAME$_cooldown }
}

target_has_max_stacks_trigger = {
	trigger_if = {
		limit = {
			flag:$SPELL_NAME$ = flag:summon_fire_elemental
		} # Max 1 stack
		$TARGET$ = { has_modifier = summon_fire_elemental_modifier }
	}
}
