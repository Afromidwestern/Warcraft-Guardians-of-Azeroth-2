MAGIC_is_shunned_or_illegal_trigger = {
	faith = {
		OR = {
			has_doctrine_parameter = $MAGIC$_magic_illegal
			has_doctrine_parameter = $MAGIC$_magic_shunned
		}
	}
}
MAGIC_is_approved_trigger = {
	faith = {
		has_doctrine_parameter = $MAGIC$_magic_approved
	}
}
