health_tree_auto_selection_weight_add_base_trigger = {
	trigger_if = {
		limit = {
			OR = {
				has_trait = weak
				has_trait = physique_bad
				health <= 3
				age >= age_45_value
				current_weight >= 50
			}
			NOT = { 
				OR = {
					has_trait = lunatic_genetic
					has_trait = lunatic_1
				} 
			}
		}
		
	}
	trigger_else = { always = no }
}
