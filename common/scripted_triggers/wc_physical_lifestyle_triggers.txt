might_tree_auto_selection_weight_add_base_trigger = {
	trigger_if = {
		limit = { has_trait = strong }
	}
	trigger_else = { always = no }
}

dexterity_tree_auto_selection_weight_add_base_trigger = {
	trigger_if = {
		limit = { 
			OR = {
				has_trait = shrewd
				is_hunter_trigger = yes
			}
		}
	}
	trigger_else = { always = no }
}

is_hunter_trigger = {
	trigger_if = {
		limit = {
			OR = {
				has_trait = hunter_1
				has_trait = hunter_2
				has_trait = hunter_3
			}
		}
	}
}

has_focus_physical_trigger = {
	OR = {
		has_focus = physical_focus_1
		has_focus = physical_focus_2
		has_focus = physical_focus_3
	}
}
