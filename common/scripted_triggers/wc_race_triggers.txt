is_humanoid_trigger = {
	custom_tooltip = {
		text = is_humanoid_trigger_tooltip
		hidden_tooltip = {
			# NOT = { has_trait = being_demon }
			OR = {
				is_small_humanoid_trigger = yes
				is_medium_humanoid_trigger = yes
				is_large_humanoid_trigger = yes
			}
		}
	}
}
