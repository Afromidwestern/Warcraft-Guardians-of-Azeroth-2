is_scourge_jailer_of_trigger = {
	trigger_if = {
		limit = {
			exists = scope:story
			scope:story = { story_type = story_scourge_invasion }
		}
	}
	trigger_else = {
		$TARGET$ = {
			any_owned_story = {
				story_type = story_scourge_invasion
				save_temporary_scope_as = story
			}
		}
	}
	
	scope:story = {
		is_target_in_variable_list = { name = nerzhul_jailors target = prev }
	}
}
any_scourge_jailer_is_avaliable_trigger = {
	trigger_if = {
		limit = {
			exists = scope:story
			scope:story = { story_type = story_scourge_invasion }
		}
	}
	trigger_else = {
		any_owned_story = {
			story_type = story_scourge_invasion
			save_temporary_scope_as = story
		}
	}
	
	scope:story = {
		any_in_list = {
			variable = nerzhul_jailors
			
			is_avaliable_scourge_jailer_trigger = yes
		}
	}
}

is_avaliable_scourge_jailer_trigger = {
	is_alive = yes
	is_imprisoned = no
}
