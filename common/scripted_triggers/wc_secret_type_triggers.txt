# SECRET PLAGUE
secret_plague_is_valid_trigger = {
	$OWNER$ = {
		NOT = {
			has_trait = deviant
		}
	}
}

secret_plague_is_shunned_trigger = {
	$OWNER$ = {
		OR = {
			religion = religion:deathbringer
			any_liege_or_above = { religion = religion:deathbringer }
		}
	}
}

secret_plague_is_criminal_trigger = {
	$OWNER$ = {
		OR = {
			religion = religion:deathbringer
			any_liege_or_above = { religion = religion:deathbringer }
		}
	}
}
# SECRET PLAGUE - END
