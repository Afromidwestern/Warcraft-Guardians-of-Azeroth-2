secret_light_user = { 
	category = religious

	is_valid = {
		can_openly_learn_light_magic = no
	}

	is_shunned = {
		light_magic_is_shunned_trigger = yes
	}

	is_criminal = {
		light_magic_is_criminal_trigger = yes
	}

	
	on_expose = {
		save_scope_as = secret
		
		secret_exposed_notification_effect = yes

		scope:secret_owner = {
			if = {
				limit = {
					NOT = {
						this = scope:secret_exposer
					}
				}
				save_scope_as = owner
				trigger_event = magic_lifestyle.7000
			}
		}
	}
}
