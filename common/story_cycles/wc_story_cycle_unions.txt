union_story_cycle = {
	on_setup = {
		#set_variable = { name = this_union value = flag:alliance }
		add_to_variable_list = {
			name = union_members
			target = story_owner
		}
	}

	on_end = {
	}

	on_owner_death = {
	}

	effect_group = {
		years = 1
		
		triggered_effect = {
			trigger = {
				var:this_union = flag:custom
				NOT = { global_var:custom_leader = story_owner }
			}
			effect = {
				set_global_variable = { name = custom_leader value = story_owner }
			}
		}
	}
}
