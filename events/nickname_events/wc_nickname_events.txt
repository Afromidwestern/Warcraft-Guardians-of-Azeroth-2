namespace = WCNIC

### the Old Murkeye
WCNIC.0001 = { # Nickname for when the character rule long
    type = character_event
	title = WCNIC.0001.t

	desc = {
		triggered_desc = {
			trigger = { has_culture_group = culture_group:murloc_group }
			desc = WCNIC.0001.desc.m
		}
		triggered_desc = {
			trigger = { has_culture_group = culture_group:human_group }
			desc = WCNIC.0001.desc.h
		}
	}

    theme = default
	override_background = {
		event_background = feast 
	}
	left_portrait = root

    trigger = {
		NOR = {
			has_character_flag = had_nickname_event
            has_character_flag = denied_nickname_event_the_old
		}
        AND = {
			#NOT = {
			#	primary_title = { recent_history = { years = 10 } }
			#}
			OR = {
				has_culture_group = culture_group:human_group
				has_culture_group = culture_group:murloc_group
			}
            age >= 45
        }
	}

    weight_multiplier = {
		base = 50
		modifier = {
			add = 2
            age >= 55
		}
        modifier = {
			add = 1
			age >= 65
		}
        #modifier = {
			#add = 5
			#NOT = {
			#	primary_title = { recent_history = { years = 20 } }
			#}
		#}
		#modifier = {
			#add = 7
			#NOT = {
			#	primary_title = { recent_history = { years = 30 } }
			#}
		 #}
	}

    immediate = {
		add_character_flag = {
			flag = had_nickname_event
		}
	}

    option = { # the Old Murkeye
		name = WCNIC.0001.a.m
        flavor = {
			desc = WCNIC.0001.tt.a.m
		}

		trigger = {
			has_culture_group = culture_group:murloc_group
		}
		
		set_nickname_effect = { NICKNAME = nick_the_old_murkeye }
		give_self_nickname_effect = yes
        add_prestige = 25
		
		ai_chance = {
			base = 100
		}
	}

	option = { # the Old
		name = WCNIC.0001.a.h

		trigger = {
			has_culture_group = culture_group:human_group
		}
		
		set_nickname_effect = { NICKNAME = nick_the_old }
		give_self_nickname_effect = yes
        add_prestige = 25
		
		ai_chance = {
			base = 100
		}
	}

    option = { # None
		name = {
			trigger = {
				has_culture_group = culture_group:murloc_group
			}

			text = WCNIC.0001.b.m
			# flavor = WCNIC.0001.tt.b.m
		}

		name = {
			trigger = {
				has_culture_group = culture_group:human_group
			}

			text = WCNIC.0001.b.h
		}
		
		remove_character_flag = had_nickname_event
        add_character_flag = {
			flag = denied_nickname_event_the_old
		}
		add_trait = humble
		
		ai_chance = {
			base = 100
		}
	}
}
