# Setup Blackhand and the associated story
WCHOI.1 = {
	type = empty
	hidden = yes
	
	trigger = {
		current_date >= 583.1.1
		NOT = { exists = global_var:dark_portal_was_opened }
	}

	immediate = {
		set_global_variable =  {
			name = dark_portal_was_opened
			value = yes
		}
		
		debug_log = "The Horde has appeared!"
		debug_log_date = yes

		title:e_horde.holder = {
			create_story = story_horde_invasion
			save_scope_as = story_owner
		}
	}
}

# Starts the Horde Invasion
WCHOI.2 = {
	type = character_event
	title = WCHOI_2_TITLE
	desc = WCHOI_2_DESC
	theme = war
	override_background = {
		reference = wc_background_dark_portal
	}
	left_portrait = {
		character = scope:blackhand
		animation = personality_bold
	}
	right_portrait = scope:stormwind_holder

	option = {
		name = {
			text = {
				first_valid = {
					triggered_desc = { # Corrupted orcs option
						trigger = { faith = faith:orcish_fel }
						desc = WCHOI_2_OPT_A
					}
					triggered_desc = { # Frostwolf option
						trigger = { faith = faith:orcish_shamanism }
						desc = WCHOI_2_OPT_B
					}
					desc = WCHOI_2_OPT_C # Outsider option
				}
			}
		}
	}
}

# Ends the Horde Invasion when enough lands taken
WCHOI.3 = {
	type = character_event
	title = WCHOI_2_TITLE
	desc = WCHOI_3_DESC
	theme = war
	override_background = {
		reference = wc_background_dark_portal
	}
	left_portrait = scope:warchief

	# The orcs need their warchief!
	option = {
		trigger = {
			OR = {
				culture_group = culture_group:orc_group
				root = scope:warchief
			}
		}
		name = WCHOI_3_OPT_A

		if = {
			limit = {
				NOT = { root = scope:warchief }
				highest_held_title_tier < scope:warchief.highest_held_title_tier
			}

			# offer_fealty_interaction
			scope:warchief = {
				save_scope_as = recipient
				add_opinion = {
					target = root
					modifier = grateful_opinion
					opinion = 30
				}
			}
			root = { save_scope_as = actor }
			offer_fealty_interaction_effect = yes
			trigger_event = char_interaction.0010
		}
		else = {
			add_character_modifier = {
				modifier = attempting_reunification_modifier
				years = 3
			}
		}
	}
	# We'll go our own way
	option = {
		trigger = {
			culture_group = culture_group:orc_group
		}
		name = WCHOI_3_OPT_B

		if = {
			limit = {
				OR = {
					root = scope:warchief
					highest_held_title_tier >= scope:warchief.highest_held_title_tier
				}
			}

			add_prestige = 1250
		}
	}
	option = {
		trigger = {
			NOT = {
				culture_group = culture_group:orc_group
				root = scope:warchief
			}
		}
		name = ITS_OVER
	}
}
