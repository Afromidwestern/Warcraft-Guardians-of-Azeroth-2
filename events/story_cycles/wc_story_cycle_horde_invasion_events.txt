# Setup Blackhand and the associated story
WCHOI.1 = {
	type = empty
	hidden = yes
	
	trigger = {
		current_date >= 583.1.1
		NOT = { exists = global_var:dark_portal_was_opened }
	}

	immediate = {
		set_global_variable =  {
			name = dark_portal_was_opened
			value = yes
		}
		
		debug_log = "The Horde has appeared!"
		debug_log_date = yes

		title:e_horde.holder = {
			create_story = story_horde_invasion
			save_scope_as = story_owner
		}
	}
}

# Starts the Horde Invasion
WCHOI.2 = {
	type = character_event
	title = WCHOI_2_TITLE
	desc = WCHOI_2_DESC
	theme = war
	override_background = {
		reference = wc_background_dark_portal
	}
	left_portrait = {
		character = scope:blackhand
		animation = personality_bold
	}
	right_portrait = scope:stormwind_holder

	option = { # Corrupted orcs option
		trigger = {
			NOT = { 
				faith = faith:orcish_shamanism
			}
			faith = faith:orcish_fel
		}
		name = WCHOI_2_OPT_A
	}

	option = { # Frostwolf option
		trigger = {
			NOT = { 
				faith = faith:orcish_fel
			}
			faith = faith:orcish_shamanism
		}
		name = WCHOI_2_OPT_B
	}
	
	option = { # Outsider option
		trigger = {
			NOT = { 
				faith = faith:orcish_shamanism
				faith = faith:orcish_fel
			}
		}
		name = WCHOI_2_OPT_C
	}
}

# Ends the Horde Invasion when enough lands taken
WCHOI.3 = {
	type = character_event
	title = WCHOI_2_TITLE
	desc = WCHOI_3_DESC
	theme = war
	override_background = {
		reference = wc_background_dark_portal
	}
	left_portrait = scope:warchief

	# The orcs need their warchief!
	option = {
		trigger = {
			OR = {
				culture_group = culture_group:orc_group
				root = scope:warchief
			}
		}
		name = WCHOI_3_OPT_A
	}
	# We'll go our own way
	option = {
		trigger = {
			culture_group = culture_group:orc_group
		}
		name = WCHOI_3_OPT_B
	}
	option = {
		trigger = {
			NOT = {
				culture_group = culture_group:orc_group
				root = scope:warchief
			}
		}
		name = ITS_OVER
	}
}

# Stormreaver holder want to exile the Frostwolf
WCHOI.1000 = {
	type = character_event
	title = WCHOI_1000_title
	desc = WCHOI_1000_desc
	theme = unfriendly
	override_background = {
		reference = throne_room_tribal
	}
	left_portrait = {
		character = scope:stormreaver_holder
		animation = personality_vengeful
	}
	immediate = {
		debug_log = "Frostwolf Exodus has begun"
		debug_log_date = yes
	}
	trigger = {
		frostwolf_exodus_d_frostwolf_trigger = yes
		frostwolf_exodus_d_stormreaver_trigger = yes
	}

	option = {
		# Ask to the Frostwolf
		name = WCHOI_1000_a
		custom_tooltip = WCHOI_1000_a_tt
		scope:frostwolf_holder = {
			trigger_event = { 
				id = WCHOI.1001
				days = 4
			}
		}
		ai_chance = {
			base = 100
		}
	}
	option = {
		# Do nothing
		name = WCHOI_1000_b
		custom_tooltip = WCHOI_1000_b_tt
	}
}

# Stormreaver letter for meeting
WCHOI.1001 = {
	type = letter_event
	opening = WCHOI.1001_opening
	desc = WCHOI.1001_desc

		character = scope:stormreaver_holder
		animation = personality_bold
	}

	trigger = {

		NOT = { root = scope:stormreaver_holder }

	option = {
		name = WCHOI.1001_a
		custom_tooltip = WCHOI.1001_a_tt

			trigger_event {
				id = WCHOI.1002
				days = 7
			}
		}
	}
}
