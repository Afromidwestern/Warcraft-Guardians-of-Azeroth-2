#Earth Mother's Blessing
tauren.2 = {
	type = character_event
	title = tauren_event_2_name
	desc = tauren.2.desc
	theme = faith
	override_background = throne_room_tribal
	
	cooldown = { years = 25 }
	
    trigger = {
        faith = faith:earth_mother_worship
		#OR{
		#	is_available_adult = yes
		#	is_available_ai_adult = yes
		#}
		AND{
			NOT { has_trait = arrogant }
			NOT { has_trait = deceitful }
			NOT { has_trait = cynical }
			NOT { has_trait = sadistic }
			NOT { has_trait = excommunicated }
			NOT { has_trait = torturer }
			NOT { has_trait = flagellant }
			NOT { has_trait = callous }
			NOT { has_trait = arbitrary }
			NOT { has_character_modifier = broke_truce }
			OR { has_trait = brave
				 has_trait = calm
				 has_trait = just
				 has_trait = compassionate
			}
			age > 45
			
			NOT { has_trait = wc_tauren_EM_wolf }
			NOT { has_trait = wc_tauren_EM_elk }
			NOT { has_trait = wc_tauren_EM_bear }
			NOT { has_trait = wc_tauren_EM_kodo }
			NOT { has_trait = wc_tauren_EM_hawk }
			
			NOT { has_trait = wc_tauren_EM_NEG_wolf }
			NOT { has_trait = wc_tauren_EM_NEG_elk }
			NOT { has_trait = wc_tauren_EM_NEG_bear }
			NOT { has_trait = wc_tauren_EM_NEG_kodo }
			NOT { has_trait = wc_tauren_EM_NEG_hawk }
		}
    }
	
 	left_portrait = {
 		character = ROOT
 		animation = ecstasy
	}
	
	option = {
		name = tauren.2.a
		trigger_event = {
			id = tauren.21
		}
	}
}
