tauren.5 = {
	type = character_event
	title = tauren_event_5_name
	desc = tauren.5.desc
	theme = skull
	override_background = wilderness
	
	left_portrait = {
 		character = ROOT
	}
	right_portrait = {
		character = scope:companion
	}
	
	#Either in Great Barrens or Elderthalas
	trigger = {
		OR = {
			title:e_great_barrens = {
				any_in_de_jure_hierarchy = {
					#tier = tier_county
					holder = root
				}
			}
			title:e_eldrethalas = {
				any_in_de_jure_hierarchy = {
					holder = root
				}
			}
		}
    }
	
	#Companion
	immediate = {
		random_knight = {
			limit = {
				AND{
					is_available_adult = yes
					#martial >= 7
					opinion = {
						target = root
						value > -40
					}
				}
			}
			save_scope_as = companion
		}
		
		if = {
			limit = {
				NOT = { exists = scope:companion }
			}
			random_courtier = {
				limit = {
					AND{
						is_available_adult = yes
						#martial >= 7
						opinion = {
							target = root
							value > -40
						}
					}
				}
				save_scope_as = companion
			}
		}

	}
	#Investigate
	option = {
		name = tauren.5.a
		
		random_neighboring_top_liege_realm_owner = {
			limit = {
				AND = {
					NOT = {
						any_war_enemy = { this = root }
						has_relation_rival = root
					}
					opinion = {
						target = root
						value > -30
					}
				}
			}
			save_scope_as = kodo_owner
		}
		
		#If no rules acceptable opinions around
		if = {
			limit = {
				NOT = { exists = scope:kodo_owner }
			}
			random_neighboring_top_liege_realm_owner = {
				limit = {
					AND = {
						NOT = {
							any_war_enemy = { this = root }
							has_relation_rival = root
						}
					}
				}
				save_scope_as = kodo_owner
			}
		}
		
		
		trigger_event = {
			id = tauren.51
		}
	}
	#Leave
	option = {
		name = tauren.5.b
	}
}

tauren.52 = {
	type = character_event
	title = tauren_event_5_name
	desc = tauren.52.desc
	theme = skull
	override_background = battlefield
		
 	left_portrait = {
 		character = ROOT
	}
	
	#Creating Bandit Character just in case
	immediate = {
		create_character = {
			template = bandit_character
			location = root.location
			culture = root.location.culture
			faith = root.location.faith
			gender = male
			save_scope_as = shoeless_bandit
		}
	}
	
	#Steal loot and let it die
	option = {
		name = tauren.52.a
		
		add_gold = 20
		if = { limit = { 					
					AND{
						faith = { NOT = { trait_is_virtue = sadistic } }
						faith = { NOT = { trait_is_virtue = callous } }
					}
			} 
				add_piety = -30
			}	
		if = { limit = { 					
					OR{
						faith = { trait_is_virtue = sadistic }
						faith = { trait_is_virtue = callous }
					}
			} 
				add_piety = 15 
		}
		every_courtier = {
			#Check if person is not evil or has a good religion that he adhers to
			limit = {  
				AND{
					NOT = { has_trait = sadistic }
					NOT = { has_trait = callous }
					OR{
						AND{
							faith = { NOT = { trait_is_virtue = sadistic } }
							faith = { NOT = { trait_is_virtue = callous } }
						}
						has_trait = cynical
					}
				}
			}
			add_opinion = {
				modifier = cruelty_opinion
				opinion = -15
				target = root
				years = 3
			}
		}
		every_courtier = {
			#Check if person is evil or has an evil religion that he adhers to
			limit = {  
				OR{
					has_trait = sadistic
					has_trait = callous
					AND{
						OR{
							faith = { trait_is_virtue = sadistic }
							faith = { trait_is_virtue = callous }
						}
						NOT = { has_trait = cynical }
					}
				}
			}
			add_opinion = {
				modifier = cruelty_opinion
				opinion = 10
				target = root
				years = 3
			}
		}
	}
	
	#Put it out of it's misery
	option = {
		name = tauren.52.b
			
		if = { limit = { 					
					AND{
						faith = { NOT = { trait_is_virtue = sadistic } }
						faith = { NOT = { trait_is_virtue = callous } }
					}
			} 
				add_piety = 20
			}	
		if = { limit = { 					
					OR{
						faith = { trait_is_virtue = sadistic }
						faith = { trait_is_virtue = callous }
					}
			} 
				add_piety = -20 
		}
		every_courtier = {
			limit = {  
				AND{
					NOT = { has_trait = sadistic }
					NOT = { has_trait = callous }
					OR{
						AND{
							faith = { NOT = { trait_is_virtue = sadistic } }
							faith = { NOT = { trait_is_virtue = callous } }
						}
						has_trait = cynical
					}
				}
			}
			add_opinion = {
				modifier = merciful_opinion
				opinion = 15
				target = root
				years = 3
			}
		}
		every_courtier = {
			limit = {  
				OR{
					has_trait = sadistic
					has_trait = callous
					AND{
						OR{
							faith = { trait_is_virtue = sadistic }
							faith = { trait_is_virtue = callous }
						}
						NOT = { has_trait = cynical }
					}
				}
			}
			add_opinion = {
				modifier = merciful_opinion
				target = root
				years = 3
				opinion = -10
			}
		}
	}
	
	#Try to heal Kodo, can either return it to owner/take it/sell it/it dies
	option = {
		name = tauren.52.c
		
		#Learning Skill Check
		duel = {
			skills = { learning }
			target = scope:shoeless_bandit
			50 = { # success
				add = {
					value = scope:duel_value
					multiplier = 3.5
					min = -49
				}
				modifier = {
					add = {
						value = 10
						if = {
							limit = { 
								has_lifestyle = learning_lifestyle
							}
							add = 10
						}
					}
				}
				modifier = {
					add = {
						value = 25
						if = {
							limit = { 
								learning > 10
								learning < 20
							}
							add = 25
						}
					}
				}
				modifier = {
					value = 55
					add = {
						if = {
							limit = { 
								learning >= 20
							}
							add = 85
						}
					}
				}
				custom_tooltip = tauren.52.c_winSkillCheck
				trigger_event = tauren.521
			}
			140 = { # failure
				add = {
					value = scope:duel_value
					multiplier = -3.5
					min = -49
				}
				custom_tooltip = tauren.52.c_loseSkillCheck
				trigger_event = tauren.522
			}
		}
	}
	
	#Sit with it till it passes
	option = {
		name = tauren.52.d
		
		trigger = {
			has_trait = compassionate
		}
		
		if = { limit = { 					
					AND{
						faith = { NOT = { trait_is_virtue = sadistic } }
						faith = { NOT = { trait_is_virtue = callous } }
					}
			} 
				add_piety = 30
			}

					OR{
						faith = { trait_is_virtue = sadistic }
						faith = { trait_is_virtue = callous }
					}
			} 
				add_piety = -25 
		}
		every_courtier = {
			#Check if person is not evil or has a good religion that he adhers to
			limit = {  
				AND{
					NOT = { has_trait = sadistic }
					NOT = { has_trait = callous }
					OR{
						AND{
							faith = { NOT = { trait_is_virtue = sadistic } }
							faith = { NOT = { trait_is_virtue = callous } }
						}
						has_trait = cynical
					}
				}
			}
			add_opinion = {
				modifier = merciful_opinion
				opinion = 25
				target = root
				years = 3
			}
		}
		every_courtier = {
			#Check if person is evil or has an evil religion that he adhers to
			limit = {  
				OR{
					has_trait = sadistic
					has_trait = callous
					AND{
						OR{
							faith = { trait_is_virtue = sadistic }
							faith = { trait_is_virtue = callous }
						}
						NOT = { has_trait = cynical }
					}
				}
			}
			add_opinion = {
				modifier = merciful_opinion
				opinion = -20
				target = root
				years = 3
			}
		}
