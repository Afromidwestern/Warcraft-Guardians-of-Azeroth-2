					}

				}
			}
		}
	}

### Beyond the Dark Portal
# Trigger event
WCDRA.995 = {
	type = empty
	hidden = yes
	
	trigger = {
		# Can't happen twice
		NOT = { is_target_in_global_variable_list = { name = unavailable_unique_events target = flag:beyond_dark_portal } }
		game_start_date < 592.1.1
		
		# Ner'zhul should be alive
		current_date > 560.1.1
		current_date < 620.1.1
		# The Dark Portal is guarded, there's no way to get to Azeroth so Ner'zhul has to open new portals
		title:c_dark_portal.holder = {
			draenor_culture_trigger = no
			draenor_faith_trigger = no
			top_liege = {
				draenor_culture_trigger = no
				draenor_faith_trigger = no
			}
		}
	}
	
	immediate = {
		add_to_global_variable_list = {
			name = unavailable_unique_events
			target = flag:beyond_dark_portal
		}
		
		title:c_dark_portal = {
			holder.top_liege = { save_temporary_scope_as = portal_top_liege }
			title_province = {
				spawn_activity = {
					type = activity_beyond_dark_portal
					owner = scope:portal_top_liege
				}
			}
		}
	}
}
# Orcs made their way through the Dark Portal and stole the magical artifacts. Everyone is worried.
WCDRA.1000 = {
	type = character_event
	title = WCDRA_1000_TITLE
	desc = WCDRA_1000_DESC
	theme = war
	override_background = {
		reference = wc_background_dark_portal
	}
	left_portrait = {
		character = root
		animation = worry
	}
	
	# Get ready for a campaign, <Knights>
	option = {
		name = WCDRA_1000_OPT_A
		
		scope:activity = {
			decline_invitation_for_character = prev
		}
		
		ordered_knight = {
			limit = {
				is_ai = yes	#Players decide themselves
				
				is_available_for_activity_trigger = yes
			}
			order_by = prowess
			max = 3
			
			add_prestige = major_prestige_gain
			
			scope:activity = {
				invite_character_to_activity = prev
				accept_invitation_for_character = prev
			}
		}
		
		ai_chance = {
			base = 100
		}
	}
	# As the <Title>, I'll lead
	option = {
		trigger = {
			is_available_for_activity_trigger = yes
		}
		name = WCDRA_1000_OPT_B
		
		scope:activity = {
			accept_invitation_for_character = prev
		}
		
		add_prestige = major_prestige_gain
		
		ordered_knight = {
			limit = {
				is_ai = yes #Players decide themselves
				
				is_available_for_activity_trigger = yes
			}
			order_by = prowess
			max = 2
			
			add_prestige = major_prestige_gain
			
			scope:activity = {
				invite_character_to_activity = prev
				accept_invitation_for_character = prev
			}
		}
		
		stress_impact = {
			brave = massive_stress_loss
			diligent = medium_stress_loss
			lazy = medium_stress_gain
			craven = massive_stress_gain
		}
		
		ai_chance = {
			base = 25 # It looks too dangerous
			
			# Stress
			modifier = {
				add = { value = 0 subtract = massive_stress_loss }
				has_trait = brave
			}
			modifier = {
				add = { value = 0 subtract = medium_stress_loss }
				has_trait = diligent
			}
			modifier = {
				add = { value = 0 subtract = medium_stress_gain }
				has_trait = lazy
			}
			modifier = {
				add = { value = 0 subtract = massive_stress_gain }
				has_trait = lazy
			}
			
			# If I'm a bad commander, I won't be needed
			modifier = {
				add = -100
				NOT = {
					has_trait_rank = {
						trait = education_martial
						rank < 2
					}
				}
			}
		}
	}
	# A bunch of bandits, not worth my attention
	option = {
		name = WCDRA_1000_OPT_C
		
		scope:activity = {
			decline_invitation_for_character = prev
		}
		
		stress_impact = {
			paranoid = minor_stress_gain
			brave = medium_stress_gain
		}
		
		ai_chance = {
			base = 25 # Ignoring doesn't look like a good idea
			
			# Stress
			modifier = {
				add = { value = 0 subtract = minor_stress_gain }
				has_trait = paranoid
			}
			modifier = {
				add = { value = 0 subtract = medium_stress_gain }
				has_trait = brave
			}
			
			# We wage our own war

				add = 100
