	option = {
		name = WCDRA_3_OPT_B
		
		add_character_flag = { flag = refused_destroy_dark_portal_flag years = 3 }
		
		ai_chance = { base = 0 }
	}
}

		reference = wc_background_dark_portal
	}

		character = scope:portal_destroyer

	}
	
	immediate = {
		show_as_tooltip = {
			destroy_dark_portal_effect = yes
		}
	}
	
	option = {
		# We used to call it home...
		name = {
			text = {
				first_valid = {
					# We used to call it home...
					triggered_desc = {
						trigger = {
							draenor_culture_trigger = yes
							draenor_faith_trigger = yes
						}
						desc = WCDRA_5_OPT_A
					}
					# What a loss.
					triggered_desc = {
						trigger = {
							OR = {
								draenor_culture_trigger = yes
								draenor_faith_trigger = yes
							}
						}
						desc = WCDRA_5_OPT_B
					}
					#And may it never open again.
					desc = WCDRA_5_OPT_C
				}
			}
		}
	}
}

# Main event
WCDRA.100 = {
	type = character_event
	title = WCDRA_100_TITLE
	desc = {
		desc = WCDRA_100_DESC
		triggered_desc = {
			trigger = { scope:refugee = { culture_group = culture_group:draenei_group } }
			desc = WCDRA_100_DESC_draenei
		}
		triggered_desc = {
			trigger = {
				scope:refugee = {
					OR = {
						culture_group = culture_group:ogre_group
						culture_group = culture_group:orc_group
					}
				}
			}
			desc = WCDRA_100_DESC_village
		}
		triggered_desc = {
			trigger = { scope:refugee = { culture_group = culture_group:arakkoa_group } }
			desc = WCDRA_100_DESC_arakkoa
		}
		desc = WCDRA_100_DESC_ending
	}
	theme = dungeon
	left_portrait = {
		character = scope:refugee
		animation = worry
	}
	
	immediate = {
		random_list = {
			5 = {
				create_character = {
					save_scope_as = refugee
					location = root.capital_province
					template = draenor_draenei_refugee
					dynasty = none
				}
				draenei_chance_modifiers = yes
			}
			10 = {
				create_character = {
					save_scope_as = refugee
					location = root.capital_province
					template = draenor_arakkoa_refugee
					dynasty = none
				}
				arakkoa_chance_modifiers = yes
			}
			10 = {
				create_character = {
					save_scope_as = refugee
					location = root.capital_province
					template = draenor_ogre_refugee
					dynasty = none
				}
				ogre_chance_modifiers = yes
			}
			20 = {
				create_character = {
					save_scope_as = refugee
					location = root.capital_province
					template = draenor_orc_refugee
					dynasty = none
				}
				orc_chance_modifiers = yes
			}
		}
		
		hidden_effect = {
			add_visiting_courtier = scope:refugee
			add_opinion = { modifier = crossed_dark_portal target = scope:refugee }
		}
	}
	
	# Let them stay
	option = {
		name = WCDRA_100_OPT_A
		
		stress_impact = {
			compassionate = minor_stress_loss
			callous = minor_stress_gain
			paranoid = minor_stress_gain
			sadistic = minor_stress_gain
		}
		
		recruit_courtier = scope:refugee
		remove_opinion = { modifier = crossed_dark_portal target = scope:refugee }
		
		ai_chance = {
			base = 200
			# Stress
			modifier = {
				add = 10
				has_trait = compassionate
			}
			modifier = {
				add = -20
				has_trait = callous
			}
			modifier = {
				add = -20
				has_trait = paranoid
			}
			modifier = {
				add = -20
				has_trait = sadistic
			}
			
			# Opinion
			opinion_modifier = {
				opinion_target = scope:refugee
				multiplier = 1
			}
		}
	}
	# Get out of my sight
	option = {
		name = WCDRA_100_OPT_B
		
		stress_impact = {
			generous = minor_stress_impact_gain
			compassionate = medium_stress_gain
		}
		
		scope:refugee = { select_and_move_to_pool_effect = yes }
		
		ai_chance = {
			base = 100
			# Stress
			modifier = {
				add = -20
				has_trait = generous
			}
			modifier = {
				add = -40
				has_trait = compassionate
			}
		}
	}
	# Guards, take them to jail
	option = {
		name = WCDRA_100_OPT_C
		
		stress_impact = {
			generous = medium_stress_gain
			compassionate = major_stress_gain
		}
		
		imprison_character_effect = {
			TARGET = scope:refugee
			IMPRISONER = root
		}
		
		ai_chance = {
			base = 100
			# Stress
			modifier = {
				add = -40
				has_trait = generous
			}
			modifier = {
				add = -60
				has_trait = compassionate
			}
		}
	}
}

# Main event
WCDRA.200 = {
	type = character_event
	title = WCDRA_200_TITLE
	desc = {
		desc = WCDRA_200_DESC_beginning
		triggered_desc = {
			trigger = { scope:scandi_adventurer = { culture_group = culture_group:draenei_group } }
			desc = WCDRA_100_DESC_draenei
		}
		triggered_desc = {
			trigger = { scope:scandi_adventurer = { culture_group = culture_group:arakkoa_group } }
			desc = WCDRA_200_DESC_arakkoa
		}
		triggered_desc = {
			trigger = { scope:scandi_adventurer = { culture_group = culture_group:ogre_group } }
			desc = WCDRA_200_DESC_ogres
		}
		triggered_desc = {
			trigger = { scope:scandi_adventurer = { culture_group = culture_group:orc_group } }
			desc = WCDRA_200_DESC_orcs
		}
		desc = WCDRA_200_DESC_ending
	}
	theme = war
	override_background = {
		reference = wc_background_dark_portal
	}
	left_portrait = {
		character = scope:scandi_adventurer
		animation = war_over_win
	}
	
	immediate = {
		# Increase commander spawn chance
		save_scope_value_as = { name = raiders value = yes }
		
		# Spawns raider
		random_list = {
			5 = {
				create_character = {
					save_scope_as = scandi_adventurer
					location = title:c_dark_portal.title_province
					template = draenor_draenei_refugee
				}
				draenei_chance_modifiers = yes
			}
			10 = {
				create_character = {
					save_scope_as = scandi_adventurer
					location = title:c_dark_portal.title_province
					template = draenor_arakkoa_refugee
				}
				arakkoa_chance_modifiers = yes
			}
			10 = {
				create_character = {
					save_scope_as = scandi_adventurer
					location = title:c_dark_portal.title_province
					template = draenor_ogre_refugee
				}
				ogre_chance_modifiers = yes
			}
			20 = {
				create_character = {
					save_scope_as = scandi_adventurer
					location = title:c_dark_portal.title_province
					template = draenor_orc_refugee
				}
				orc_chance_modifiers = yes
			}
		}
		# Adds traits
		scope:scandi_adventurer = {
			add_character_modifier = scandinavian_adventurer_fp1_modifier
			add_trait = adventurer
		}
	}
	
	# Settle down and let them through peacefully.
	option = {
		name = WCDRA_200_OPT_A
		
		# Targets random ruler
		add_weak_ruler_adventurer_target_effect = { REGION = world_eastern_kingdoms }
		
		ai_chance = {
			base = 0
			# Draenor faiths and cultures may let them come
			modifier = {
				add = 500
				draenor_culture_trigger = yes
			}
			modifier = {
				add = 500
				draenor_faith_trigger = yes
			}
		}
	}
	# Only over my dead body.
	option = {
		trigger = {
			valid_adventurer_target_trigger = yes
		}
		name = WCDRA_200_OPT_B
		
		# Targets you
		add_ruler_adventurer_target_effect = yes
		
		stress_impact = {
			craven = minor_stress_gain
			compassionate = minor_stress_gain
			generous = minor_stress_gain
		}
		
		ai_chance = {
			base = 100
			# Stress
			modifier = {
				add = -20
				has_trait = craven
			}
			modifier = {
				add = -20
				has_trait = compassionate
			}
			modifier = {
				add = -20
				has_trait = generous
			}
		}
	}
	
	after = {
		# Declares war
		if = {
			limit = { exists = scope:adventurer_target }
			scope:scandi_adventurer = {
				draenor_adventurer_start_war_effect = yes
			}
		}
		else = {
			scope:scandi_adventurer = { remove_character_modifier = scandinavian_adventurer_fp1_modifier }
		}
	}
}
