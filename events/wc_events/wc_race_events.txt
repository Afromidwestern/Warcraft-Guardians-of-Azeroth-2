#Assigns race traits at the game start
WCRAC.1 = {
	type = empty
	hidden = yes

	immediate = {
		every_living_character = {
			limit = { has_race_trait_trigger = no }
			trigger_event = WCRAC.5
		}
		debug_log = "WCRAC.1 fired"
	}
}
