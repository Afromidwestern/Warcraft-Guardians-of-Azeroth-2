namespace = WCTEST

#Discovers start-up techs
WCTEST.1 = {
	type = empty
	hidden = yes

	immediate = {
		every_independent_ruler = {
			limit = {
				OR = {
					culture = { has_giant_maa_trigger = yes }
					culture = { has_massive_aerial_maa_trigger = yes }
				}
			}
			if = {
				limit = { culture = { has_giant_maa_trigger = yes } }
				culture = { add_innovation = innovation_giants }
			}
			if = {
				limit = { culture = { has_massive_aerial_maa_trigger = yes } }
				culture = { add_innovation = innovation_flying_beasts }
			}
		}
	}
}
