namespace = WCTEST

#Discovers start-up techs
WCTEST.1 = {
	type = empty
	hidden = yes

	immediate = {
		every_independent_ruler = {
			limit = {
				OR = {
					###IF CULTURES HAVE MAA'S
					culture = {
						has_giant_maa_trigger = yes
						NOT = { has_innovation = innovation_giants }
					}
					culture = {
						has_massive_aerial_maa_trigger = yes
						NOT = { has_innovation = innovation_flying_beasts }
					}
					###RACE SIZE TECHS
					culture = {
						is_giant_culture_trigger = yes
						NOT = { has_innovation = innovation_massive_soldiers }
					}
					culture = {
						is_large_culture_trigger = yes
						NOT = { has_innovation = innovation_large_soldiers }
					}
					culture = {
						is_small_culture_trigger = yes
						NOT = { has_innovation = innovation_small_soldiers }
					}
				}
			}
			###MAA'S
			if = {
				limit = {
					culture = {
						has_giant_maa_trigger = yes
						NOT = { has_innovation = innovation_giants }
					}
				}
				culture = { add_innovation = innovation_giants }
			}
			if = {
				limit = {
					culture = {
						has_massive_aerial_maa_trigger = yes
						NOT = { has_innovation = innovation_flying_beasts }
					}
				}
				culture = { add_innovation = innovation_flying_beasts }
			}
			###RACE SIZE 
			if = {
				limit = {
					culture = {
						is_giant_culture_trigger = yes
						NOT = { has_innovation = innovation_massive_soldiers }
					}
				}
				culture = { add_innovation = innovation_massive_soldiers }
			}
			if = {
				limit = {
					culture = {
						is_large_culture_trigger = yes
						NOT = { has_innovation = innovation_large_soldiers }
					}
				}
				culture = { add_innovation = innovation_large_soldiers }
			}
			if = {
				limit = {
					culture = {
						is_small_culture_trigger = yes
						NOT = { has_innovation = innovation_small_soldiers }
					}
				}
				culture = { add_innovation = innovation_small_soldiers }
			}
		}
	}
}
